cdiff(confkeys)

returns a pound (’#’) char, unless @conf and $config values differ

[Source]

    # File lib/conf/wizard.rb, line 38
38:   def cdiff(confkeys)
39:     conf1 = @conf
40:     conf2 = $config
41:     # recurse into config hash
42:     confkeys.each do |confkey|
43:       conf1 = conf1[confkey]
44:       conf2 = conf2[confkey]
45:     end
46:     # return '' if the values differ, '#' if they are the same
47:     #puts "#{conf1.inspect} vs #{conf2.inspect}"
48:     if conf1 != conf2
49:       return ''
50:     else
51:       return '#'
52:     end
